def fig_to_base64_png(fig, dpi=200):
    """
    Convert a matplotlib Figure to a Base64-encoded PNG string (no header prefix).
    """
    buf = io.BytesIO()
    fig.savefig(buf, format="png", dpi=dpi, bbox_inches="tight")
    buf.seek(0)
    b64 = base64.b64encode(buf.getvalue()).decode("utf-8")
    plt.close(fig)
    return b64
